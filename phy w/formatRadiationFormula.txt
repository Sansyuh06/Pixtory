GLuint compileShader(const char* source, GLenum type) {
    GLuint shader = glCreateShader(type);
    if (!shader) throw std::runtime_error("Failed to create shader");
    glShaderSource(shader, 1, &source, nullptr);
    glCompileShader(shader);
    GLint success;
    glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
    if (!success) {
        char infoLog[512];
        glGetShaderInfoLog(shader, 512, nullptr, infoLog);
        std::cerr << "Shader compilation error (" << (type == GL_VERTEX_SHADER ? "Vertex" : "Fragment")
            << ") at line " << __LINE__ << ": " << infoLog << std::endl;
        glDeleteShader(shader);
        return 0;
    }
    return shader;
}

GLuint createShaderProgram(const char* vertexSource, const char* fragmentSource) {
    GLuint vertexShader = compileShader(vertexSource, GL_VERTEX_SHADER);
    GLuint fragmentShader = compileShader(fragmentSource, GL_FRAGMENT_SHADER);
    if (!vertexShader || !fragmentShader) return 0;

    GLuint program = glCreateProgram();
    if (!program) throw std::runtime_error("Failed to create shader program");
    glAttachShader(program, vertexShader);
    glAttachShader(program, fragmentShader);
    glLinkProgram(program);
    GLint success;
    glGetProgramiv(program, GL_LINK_STATUS, &success);
    if (!success) {
        char infoLog[512];
        glGetProgramInfoLog(program, 512, nullptr, infoLog);
        std::cerr << "Shader program linking error at line " << __LINE__ << ": " << infoLog << std::endl;
        glDeleteProgram(program);
        return 0;
    }
    glDeleteShader(vertexShader);
    glDeleteShader(fragmentShader);
    return program;
}

bool initTextRendering() {
    const int texWidth = 16 * FONT_WIDTH;
    const int texHeight = 8 * FONT_HEIGHT;
    std::vector<uint8_t> textureData(texWidth * texHeight, 0);
    if (textureData.size() != static_cast<size_t>(texWidth * texHeight)) {
        std::cerr << "Texture data size mismatch at 01:33 AM IST, August 05, 2025" << std::endl;
        return false;
    }

    for (const auto& pair : fontData) {
        char c = pair.first;
        const std::vector<uint8_t>& bitmap = pair.second;
        if (bitmap.size() != FONT_HEIGHT) {
            std::cerr << "Font bitmap size error for '" << c << "' at 01:33 AM IST, August 05, 2025" << std::endl;
            continue;
        }
        int charX = (c % 16) * FONT_WIDTH;
        int charY = (c / 16) * FONT_HEIGHT;
        for (int y = 0; y < FONT_HEIGHT; ++y) {
            uint8_t row = bitmap[y];
            for (int x = 0; x < FONT_WIDTH; ++x) {
                int idx = (charY + y) * texWidth + (charX + x);
                if (idx >= static_cast<int>(textureData.size())) {
                    std::cerr << "Texture index out of bounds at 01:33 AM IST, August 05, 2025" << std::endl;
                    return false;
                }
                textureData[idx] = (row & (1 << (FONT_WIDTH - 1 - x))) ? 255 : 0;
            }
        }
    }

    glGenTextures(1, &fontTexture);
    glBindTexture(GL_TEXTURE_2D, fontTexture);
    glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, texWidth, texHeight, 0, GL_RED, GL_UNSIGNED_BYTE, textureData.data());
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
    if (glGetError() != GL_NO_ERROR) {
        std::cerr << "OpenGL texture error at 01:33 AM IST, August 05, 2025" << std::endl;
        return false;
    }

    glGenVertexArrays(1, &textVAO);
    glGenBuffers(1, &textVBO);
    glBindVertexArray(textVAO);
    glBindBuffer(GL_ARRAY_BUFFER, textVBO);
    glBufferData(GL_ARRAY_BUFFER, sizeof(float) * 6 * 4, nullptr, GL_DYNAMIC_DRAW);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4 * sizeof(float), nullptr);
    glBindBuffer(GL_ARRAY_BUFFER, 0);
    glBindVertexArray(0);

    return true;
}

std::string formatRadiationFormula(float time, const glm::vec3& r) {
    std::stringstream ss;
    float distance = glm::length(r);
    ss << "E_rad = (-q)/(4pe_0c^2) * 1/|r| * a_perp(t - |r|/c)";
    ss << "\n|r| = " << std::fixed << std::setprecision(2) << distance;
    ss << ", t - |r|/c = " << std::setprecision(2) << (time - distance / SPEED_OF_LIGHT);
    return ss.str();
}

void renderText(GLuint shader, const std::string& text, float x, float y, float scale, const glm::vec3& color) {
    glUseProgram(shader);
    GLint textColorLoc = glGetUniformLocation(shader, "textColor");
    if (textColorLoc == -1) {
        std::cerr << "Uniform 'textColor' not found at 01:33 AM IST, August 05, 2025" << std::endl;
        return;
    }
    glUniform3fv(textColorLoc, 1, &color[0]);
    glActiveTexture(GL_TEXTURE0);
    glBindTexture(GL_TEXTURE_2D, fontTexture);
    glBindVertexArray(textVAO);

    glm::mat4 projection = glm::ortho(0.0f, static_cast<float>(WIDTH), 0.0f, static_cast<float>(HEIGHT));
    GLint projLoc = glGetUniformLocation(shader, "projection");
    if (projLoc == -1) {
        std::cerr << "Uniform 'projection' not found at 01:33 AM IST, August 05, 2025" << std::endl;
        return;
    }
    glUniformMatrix4fv(projLoc, 1, GL_FALSE, &projection[0][0]);

    float startX = x;
    for (char c : text) {
        if (c == '\n') {
            y -= 30 * scale;
            x = startX;
            continue;
        }
        auto it = fontData.find(c);
        if (it == fontData.end()) c = ' ';

        float w = 20 * scale;
        float h = 30 * scale;
        float texX = (c % 16) / 16.0f;
        float texY = (c / 16) / 8.0f;
        float texW = 1.0f / 16.0f;
        float texH = 1.0f / 8.0f;

        float vertices[6][4] = {
            {x, y + h, texX, texY}, {x, y, texX, texY + texH}, {x + w, y, texX + texW, texY + texH},
            {x, y + h, texX, texY}, {x + w, y, texX + texW, texY + texH}, {x + w, y + h, texX + texW, texY}
        };

        glBindBuffer(GL_ARRAY_BUFFER, textVBO);
        glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices);
        glDrawArrays(GL_TRIANGLES, 0, 6);
        x += w * 0.6f;
    }

    glBindVertexArray(0);
    glBindTexture(GL_TEXTURE_2D, 0);
}

void generateSphere(std::vector<float>& vertices, std::vector<unsigned int>& indices, float radius, int slices, int stacks) {
    vertices.clear();
    indices.clear();
    for (int i = 0; i <= stacks; ++i) {
        float phi = glm::pi<float>() * i / stacks;
        for (int j = 0; j <= slices; ++j) {
            float theta = 2.0f * glm::pi<float>() * j / slices;
            float x = radius * sin(phi) * cos(theta);
            float y = radius * sin(phi) * sin(theta);
            float z = radius * cos(phi);
            vertices.push_back(x);
            vertices.push_back(y);
            vertices.push_back(z);
        }
    }
    for (int i = 0; i < stacks; ++i) {
        for (int j = 0; j < slices; ++j) {
            int first = i * (slices + 1) + j;
            int second = first + slices + 1;
            int totalVertices = static_cast<int>((stacks + 1) * (slices + 1));
            if (first >= 0 && second < totalVertices && first + 1 < totalVertices &&
                second + 1 < totalVertices && j < slices - 1) {
                indices.push_back(first);
                indices.push_back(second);
                indices.push_back(first + 1);
                indices.push_back(second);
                indices.push_back(second + 1);
                indices.push_back(first + 1);
            }
        }
    }
}

void generateGrid(std::vector<float>& vertices, std::vector<unsigned int>& indices, float size, float spacing) {
    vertices.clear();
    indices.clear();
    unsigned int index = 0;
    for (float z = -size; z <= size; z += spacing) {
        vertices.push_back(-size); vertices.push_back(0.0f); vertices.push_back(z);
        vertices.push_back(size); vertices.push_back(0.0f); vertices.push_back(z);
        indices.push_back(index); indices.push_back(index + 1);
        index += 2;
    }
    for (float x = -size; x <= size; x += spacing) {
        vertices.push_back(x); vertices.push_back(0.0f); vertices.push_back(-size);
        vertices.push_back(x); vertices.push_back(0.0f); vertices.push_back(size);
        indices.push_back(index); indices.push_back(index + 1);
        index += 2;
    }
}

void generateArrow(std::vector<float>& vertices, std::vector<unsigned int>& indices) {
    vertices.clear();
    indices.clear();
    vertices.push_back(0.0f); vertices.push_back(0.0f); vertices.push_back(0.0f);
    vertices.push_back(0.0f); vertices.push_back(0.0f); vertices.push_back(1.0f);
    float headRadius = 0.05f;
    float headLength = 0.2f;
    int segments = 6;
    for (int i = 0; i < segments; ++i) {
        float angle = 2.0f * glm::pi<float>() * i / segments;
        float x = headRadius * cos(angle);
        float y = headRadius * sin(angle);
        vertices.push_back(x); vertices.push_back(y); vertices.push_back(1.0f - headLength);
    }
    indices.push_back(0); indices.push_back(1);
    for (int i = 0; i < segments; ++i) {
        indices.push_back(1); indices.push_back(2 + i);
        if (i + 1 < segments) {
            indices.push_back(2 + i);
            indices.push_back(2 + ((i + 1) % segments));
        }
    }
}

glm::vec3 calculateRadiationField(const glm::vec3& point, float time) {
    if (charges.empty()) return glm::vec3(0.0f);
    const Charge& charge = charges[0];
    glm::vec3 r = point - charge.position;
    float distance = glm::length(r);
    if (distance < 0.1f || distance == 0.0f) return glm::vec3(0.0f);

    // Slow down the wave propagation
    float delay = distance / (SPEED_OF_LIGHT * 0.3f); // Much slower wave speed
    float pastTime = time * 0.5f - delay; // Slower oscillation

    // Create electromagnetic wave pattern
    float omega = 1.0f; // Lower frequency
    glm::vec3 accel = charge.acceleration * sin(omega * pastTime);
    glm::vec3 rHat = glm::normalize(r);
    glm::vec3 accelPerp = accel - glm::dot(accel, rHat) * rHat;

    // Add wave amplitude falloff with distance
    float amplitude = 1.0f / (1.0f + distance * 0.2f);
    float magnitude = amplitude * CHARGE_Q / (4.0f * glm::pi<float>() * EPSILON_0 * SPEED_OF_LIGHT * SPEED_OF_LIGHT) * 1e4f;

    return magnitude * accelPerp;
}

void processInput(GLFWwindow* window) {
    if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
        glfwSetWindowShouldClose(window, true);

    static bool cPressed = false;
    if (glfwGetKey(window, GLFW_KEY_C) == GLFW_PRESS && !cPressed) {
        freeCameraMode = !freeCameraMode;
        autoRotate = false;
        std::cout << "Camera mode switched to " << (freeCameraMode ? "FREE" : "ORBIT") << " at 01:33 AM IST, August 05, 2025" << std::endl;
        cPressed = true;
    }
    else if (glfwGetKey(window, GLFW_KEY_C) == GLFW_RELEASE) {
        cPressed = false;
    }

    static bool spacePressed = false;
    if (glfwGetKey(window, GLFW_KEY_SPACE) == GLFW_PRESS && !spacePressed) {
        if (!freeCameraMode) autoRotate = !autoRotate;
        spacePressed = true;
    }
    else if (glfwGetKey(window, GLFW_KEY_SPACE) == GLFW_RELEASE) {
        spacePressed = false;
    }

    float speed = 0.02f;
    if (freeCameraMode) {
        glm::vec3 front = glm::normalize(glm::vec3(
            cos(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch)),
            sin(glm::radians(cameraPitch)),
            sin(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch))
        ));
        glm::vec3 right = glm::normalize(glm::cross(front, glm::vec3(0.0f, 1.0f, 0.0f)));
        glm::vec3 up = glm::normalize(glm::cross(right, front));
        float cameraSpeed = 0.1f;
        if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS) cameraPos += front * cameraSpeed;
        if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS) cameraPos -= front * cameraSpeed;
        if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS) cameraPos -= right * cameraSpeed;
        if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS) cameraPos += right * cameraSpeed;
        if (glfwGetKey(window, GLFW_KEY_Q) == GLFW_PRESS) cameraPos -= up * cameraSpeed;
        if (glfwGetKey(window, GLFW_KEY_E) == GLFW_PRESS) cameraPos += up * cameraSpeed;
        if (!charges.empty()) {
            if (glfwGetKey(window, GLFW_KEY_UP) == GLFW_PRESS) charges[0].position.z -= speed;
            if (glfwGetKey(window, GLFW_KEY_DOWN) == GLFW_PRESS) charges[0].position.z += speed;
            if (glfwGetKey(window, GLFW_KEY_LEFT) == GLFW_PRESS) charges[0].position.x -= speed;
            if (glfwGetKey(window, GLFW_KEY_RIGHT) == GLFW_PRESS) charges[0].position.x += speed;
        }
    }
    else {
        if (!charges.empty()) {
            if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS) charges[0].position.z -= speed;
            if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS) charges[0].position.z += speed;
            if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS) charges[0].position.x -= speed;
            if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS) charges[0].position.x += speed;
            if (glfwGetKey(window, GLFW_KEY_Q) == GLFW_PRESS) charges[0].position.y += speed;
            if (glfwGetKey(window, GLFW_KEY_E) == GLFW_PRESS) charges[0].position.y -= speed;
        }
        float cameraSpeed = 1.0f;
        if (glfwGetKey(window, GLFW_KEY_UP) == GLFW_PRESS) {
            autoRotate = false;
            cameraPitch = std::min(cameraPitch + cameraSpeed, 89.0f);
        }
        if (glfwGetKey(window, GLFW_KEY_DOWN) == GLFW_PRESS) {
            autoRotate = false;
            cameraPitch = std::max(cameraPitch - cameraSpeed, -89.0f);
        }
        if (glfwGetKey(window, GLFW_KEY_LEFT) == GLFW_PRESS) {
            autoRotate = false;
            cameraYaw -= cameraSpeed;
        }
        if (glfwGetKey(window, GLFW_KEY_RIGHT) == GLFW_PRESS) {
            autoRotate = false;
            cameraYaw += cameraSpeed;
        }
    }
}

void mouseCallback(GLFWwindow* window, double xpos, double ypos) {
    if (mouseButtonPressed) {
        autoRotate = false;
        float xoffset = static_cast<float>(xpos - lastMouseX);
        float yoffset = static_cast<float>(lastMouseY - ypos);
        lastMouseX = xpos;
        lastMouseY = ypos;
        float sensitivity = 0.1f;
        cameraYaw += xoffset * sensitivity;
        cameraPitch += yoffset * sensitivity;
        cameraPitch = std::max(-89.0f, std::min(cameraPitch, 89.0f));
    }
}

void mouseButtonCallback(GLFWwindow* window, int button, int action, int mods) {
    if (button == GLFW_MOUSE_BUTTON_LEFT) {
        mouseButtonPressed = (action == GLFW_PRESS);
        if (mouseButtonPressed) glfwGetCursorPos(window, &lastMouseX, &lastMouseY);
    }
}

void scrollCallback(GLFWwindow* window, double xoffset, double yoffset) {
    if (!freeCameraMode) {
        cameraDistance -= static_cast<float>(yoffset) * 0.5f;
        cameraDistance = std::max(2.0, std::min(cameraDistance, 20.0));
    }
}

int main() {
    if (!glfwInit()) {
        std::cerr << "GLFW initialization failed at 01:33 AM IST, August 05, 2025" << std::endl;
        return -1;
    }

    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
    glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE); // For macOS compatibility

    GLFWwindow* window = glfwCreateWindow(WIDTH, HEIGHT, "Radiation Field Visualization", nullptr, nullptr);
    if (!window) {
        std::cerr << "Window creation failed at 01:33 AM IST, August 05, 2025" << std::endl;
        glfwTerminate();
        return -1;
    }

    glfwMakeContextCurrent(window);
    glfwSetCursorPosCallback(window, mouseCallback);
    glfwSetMouseButtonCallback(window, mouseButtonCallback);
    glfwSetScrollCallback(window, scrollCallback);

    if (glewInit() != GLEW_OK) {
        std::cerr << "GLEW initialization failed at 01:33 AM IST, August 05, 2025" << std::endl;
        glfwTerminate();
        return -1;
    }

    glEnable(GL_DEPTH_TEST);
    glEnable(GL_BLEND);
    glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);

    if (!initTextRendering()) {
        std::cerr << "Text rendering initialization failed at 01:33 AM IST, August 05, 2025" << std::endl;
        glfwTerminate();
        return -1;
    }

    GLuint lineShader = createShaderProgram(vertexShaderSource, fragmentShaderSource);
    GLuint sphereShader = createShaderProgram(sphereVertexShaderSource, sphereFragmentShaderSource);
    GLuint textShader = createShaderProgram(textVertexShaderSource, textFragmentShaderSource);
    if (!lineShader || !sphereShader || !textShader) {
        std::cerr << "Shader program creation failed at 01:33 AM IST, August 05, 2025" << std::endl;
        glfwTerminate();
        return -1;
    }

    charges.emplace_back(glm::vec3(0.0f, 0.0f, 0.0f), CHARGE_Q, glm::vec3(1.0f, 0.3f, 0.3f), glm::vec3(0.0f, 0.0f, 1.0f));

    std::vector<float> sphereVertices, gridVertices, arrowVertices;
    std::vector<unsigned int> sphereIndices, gridIndices, arrowIndices;
    generateSphere(sphereVertices, sphereIndices, 0.2f, 20, 20);
    generateGrid(gridVertices, gridIndices, 5.0f, 0.5f);
    generateArrow(arrowVertices, arrowIndices);

    GLuint sphereVAO, sphereVBO, sphereEBO;
    glGenVertexArrays(1, &sphereVAO);
    glGenBuffers(1, &sphereVBO);
    glGenBuffers(1, &sphereEBO);
    glBindVertexArray(sphereVAO);
    glBindBuffer(GL_ARRAY_BUFFER, sphereVBO);
    glBufferData(GL_ARRAY_BUFFER, sphereVertices.size() * sizeof(float), sphereVertices.data(), GL_STATIC_DRAW);
    glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, sphereEBO);
    glBufferData(GL_ELEMENT_ARRAY_BUFFER, sphereIndices.size() * sizeof(unsigned int), sphereIndices.data(), GL_STATIC_DRAW);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), nullptr);

    GLuint gridVAO, gridVBO, gridEBO;
    glGenVertexArrays(1, &gridVAO);
    glGenBuffers(1, &gridVBO);
    glGenBuffers(1, &gridEBO);
    glBindVertexArray(gridVAO);
    glBindBuffer(GL_ARRAY_BUFFER, gridVBO);
    glBufferData(GL_ARRAY_BUFFER, gridVertices.size() * sizeof(float), gridVertices.data(), GL_STATIC_DRAW);
    glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, gridEBO);
    glBufferData(GL_ELEMENT_ARRAY_BUFFER, gridIndices.size() * sizeof(unsigned int), gridIndices.data(), GL_STATIC_DRAW);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), nullptr);

    GLuint arrowVAO, arrowVBO, arrowEBO;
    glGenVertexArrays(1, &arrowVAO);
    glGenBuffers(1, &arrowVBO);
    glGenBuffers(1, &arrowEBO);
    glBindVertexArray(arrowVAO);
    glBindBuffer(GL_ARRAY_BUFFER, arrowVBO);
    glBufferData(GL_ARRAY_BUFFER, arrowVertices.size() * sizeof(float), arrowVertices.data(), GL_STATIC_DRAW);
    glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, arrowEBO);
    glBufferData(GL_ELEMENT_ARRAY_BUFFER, arrowIndices.size() * sizeof(unsigned int), arrowIndices.data(), GL_STATIC_DRAW);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), nullptr);

    glBindVertexArray(0);

    while (!glfwWindowShouldClose(window)) {
        processInput(window);
        simulationTime += 0.008f; // Much slower time progression

        if (freeCameraMode) {
            glm::vec3 front = glm::normalize(glm::vec3(
                cos(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch)),
                sin(glm::radians(cameraPitch)),
                sin(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch))
            ));
            view = glm::lookAt(cameraPos, cameraPos + front, glm::vec3(0.0f, 1.0f, 0.0f));
        }
        else {
            if (autoRotate) cameraYaw += 0.1f; // Much slower auto rotation
            cameraPos = glm::vec3(
                static_cast<float>(cameraDistance * cos(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch))),
                static_cast<float>(cameraDistance * sin(glm::radians(cameraPitch))),
                static_cast<float>(cameraDistance * sin(glm::radians(cameraYaw)) * cos(glm::radians(cameraPitch)))
            );
            view = glm::lookAt(cameraPos, cameraTarget, glm::vec3(0.0f, 1.0f, 0.0f));
        }

        glm::mat4 projection = glm::perspective(glm::radians(45.0f), (float)WIDTH / (float)HEIGHT, 0.1f, 100.0f);
        glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
        glClearColor(0.05f, 0.05f, 0.1f, 1.0f);

        glUseProgram(lineShader);
        glm::mat4 mvp = projection * view;
        glUniformMatrix4fv(glGetUniformLocation(lineShader, "mvp"), 1, GL_FALSE, &mvp[0][0]);
        glUniform3f(glGetUniformLocation(lineShader, "color"), 0.3f, 0.3f, 0.3f);
        glUniform1f(glGetUniformLocation(lineShader, "alpha"), 0.5f);

        glBindVertexArray(gridVAO);
        glDrawElements(GL_LINES, static_cast<GLsizei>(gridIndices.size()), GL_UNSIGNED_INT, nullptr);

        const float gridSpacing = 0.6f;
        const float gridSize = 4.0f;
        const float maxArrowLength = 0.4f;

        glBindVertexArray(arrowVAO);
        for (float x = -gridSize; x <= gridSize; x += gridSpacing) {
            for (float z = -gridSize; z <= gridSize; z += gridSpacing) {
                for (float y = -1.5f; y <= 1.5f; y += gridSpacing) {
                    glm::vec3 point(x, y, z);
                    glm::vec3 field = calculateRadiationField(point, simulationTime);
                    float magnitude = glm::length(field);
                    if (magnitude > 0.001f) {
                        glm::vec3 direction = glm::normalize(field);

                        // Much slower, smoother length animation
                        float distance = glm::length(point - charges[0].position);
                        float wavePhase = simulationTime * 0.3f - distance * 0.5f; // Slow wave propagation
                        float targetLength = magnitude * 0.8f;
                        float smoothedLength = targetLength * (0.5f + 0.5f * sin(wavePhase));
                        smoothedLength = std::min(smoothedLength, maxArrowLength);

                        if (smoothedLength > 0.02f) { // Only draw visible arrows
                            glm::mat4 model = glm::translate(glm::mat4(1.0f), point);
                            glm::vec3 up(0.0f, 0.0f, 1.0f);
                            if (glm::length(glm::cross(up, direction)) > 0.001f) {
                                glm::vec3 axis = glm::normalize(glm::cross(up, direction));
                                float angle = acos(glm::clamp(glm::dot(up, direction), -1.0f, 1.0f));
                                model = glm::rotate(model, angle, axis);
                            }
                            model = glm::scale(model, glm::vec3(1.0f, 1.0f, smoothedLength));

                            mvp = projection * view * model;
                            glUniformMatrix4fv(glGetUniformLocation(lineShader, "mvp"), 1, GL_FALSE, &mvp[0][0]);

                            // Color based on wave phase - like electromagnetic spectrum
                            float colorPhase = wavePhase + distance * 0.3f;
                            float r = 0.3f + 0.7f * (0.5f + 0.5f * sin(colorPhase));
                            float g = 0.2f + 0.6f * (0.5f + 0.5f * sin(colorPhase + 2.0f));
                            float b = 0.4f + 0.6f * (0.5f + 0.5f * sin(colorPhase + 4.0f));

                            glUniform3f(glGetUniformLocation(lineShader, "color"), r, g, b);

                            // Smooth alpha with wave propagation
                            float distanceFactor = 1.0f / (1.0f + distance * 0.1f);
                            float alpha = 0.8f * distanceFactor * (0.3f + 0.7f * sin(wavePhase));
                            alpha = std::max(0.05f, std::min(0.9f, alpha));
                            glUniform1f(glGetUniformLocation(lineShader, "alpha"), alpha);

                            glDrawElements(GL_LINES, static_cast<GLsizei>(arrowIndices.size()), GL_UNSIGNED_INT, nullptr);
                        }
                    }
                }
            }
        }

        glUseProgram(sphereShader);
        glUniform3f(glGetUniformLocation(sphereShader, "lightPos"), 5.0f, 5.0f, 5.0f);
        glUniform3f(glGetUniformLocation(sphereShader, "viewPos"), cameraPos.x, cameraPos.y, cameraPos.z);
        glUniform1f(glGetUniformLocation(sphereShader, "time"), simulationTime);

        glBindVertexArray(sphereVAO);
        for (const auto& charge : charges) {
            glm::mat4 model = glm::translate(glm::mat4(1.0f), charge.position + glm::vec3(0.0f, 0.0f, 0.2f * sin(0.8f * simulationTime))); // Slower charge oscillation
            mvp = projection * view * model;
            glUniformMatrix4fv(glGetUniformLocation(sphereShader, "mvp"), 1, GL_FALSE, &mvp[0][0]);
            glUniformMatrix4fv(glGetUniformLocation(sphereShader, "model"), 1, GL_FALSE, &model[0][0]);
            glUniform3fv(glGetUniformLocation(sphereShader, "chargeColor"), 1, &charge.color[0]);
            glDrawElements(GL_TRIANGLES, static_cast<GLsizei>(sphereIndices.size()), GL_UNSIGNED_INT, nullptr);
        }

        glDisable(GL_DEPTH_TEST);

        std::string instructions = "Controls:\n";
        instructions += freeCameraMode ? "WASD/QE: Move camera\nArrow keys: Move charge" : "WASD/QE: Move charge\nArrow keys: Rotate camera";
        instructions += "\nC: Toggle camera mode\nSpace: " + std::string(freeCameraMode ? "N/A" : (autoRotate ? "Stop rotation" : "Auto rotate"));
        instructions += "\nMouse: Look around\nScroll: Zoom";
        renderText(textShader, instructions, 20.0f, HEIGHT - 40.0f, 0.6f, glm::vec3(0.8f, 0.8f, 0.8f));

        if (!charges.empty()) {
            renderText(textShader, formatRadiationFormula(simulationTime, charges[0].position), 20.0f, HEIGHT - 80.0f, 0.5f, glm::vec3(0.9f, 0.9f, 0.3f));
        }

        std::string mode = "Mode: " + std::string(freeCameraMode ? "FREE CAMERA" : "ORBIT") + (!freeCameraMode && autoRotate ? " (AUTO)" : "");
        renderText(textShader, mode, 20.0f, 40.0f, 0.6f, glm::vec3(0.3f, 0.8f, 0.3f));

        glEnable(GL_DEPTH_TEST);
        glfwSwapBuffers(window);
        glfwPollEvents();
    }

    glDeleteVertexArrays(1, &sphereVAO);
    glDeleteBuffers(1, &sphereVBO);
    glDeleteBuffers(1, &sphereEBO);
    glDeleteVertexArrays(1, &gridVAO);
    glDeleteBuffers(1, &gridVBO);
    glDeleteBuffers(1, &gridEBO);
    glDeleteVertexArrays(1, &arrowVAO);
    glDeleteBuffers(1, &arrowVBO);
    glDeleteBuffers(1, &arrowEBO);
    glDeleteVertexArrays(1, &textVAO);
    glDeleteBuffers(1, &textVBO);
    glDeleteTextures(1, &fontTexture);
    glDeleteProgram(lineShader);
    glDeleteProgram(sphereShader);
    glDeleteProgram(textShader);

    glfwTerminate();
    return 0;
}
