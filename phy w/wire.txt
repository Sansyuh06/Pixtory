#include <GL/glew.h>
#include <GLFW/glfw3.h>
#include <glm/glm.hpp>
#include <glm/gtc/matrix_transform.hpp>
#include <glm/gtc/type_ptr.hpp>
#include <imgui.h>
#include <imgui_impl_glfw.h>
#include <imgui_impl_opengl3.h>
#include <vector>
#include <cmath>
#include <iostream>

const char* vertexShaderSource = R"(
#version 330 core
layout(location = 0) in vec3 aPos;
layout(location = 1) in vec3 aColor;
out vec3 fragColor;
uniform mat4 mvp;
void main() {
    gl_Position = mvp * vec4(aPos, 1.0);
    fragColor = aColor;
}
)";

const char* fragmentShaderSource = R"(
#version 330 core
in vec3 fragColor;
out vec4 FragColor;
void main() {
    FragColor = vec4(fragColor, 1.0);
}
)";

struct Vertex {
    glm::vec3 position;
    glm::vec3 color;
};

std::vector<Vertex> createWirePath(float radius, float length) {
    std::vector<Vertex> vertices;
    const int segments = 20; // Number of segments around cylinder
    const int lengthSegments = 50; // Number of segments along length
    for (int j = 0; j < lengthSegments; ++j) {
        float z = (j / (float)(lengthSegments - 1)) * length - length / 2.0f; // Center at origin
        float zNext = ((j + 1) / (float)(lengthSegments - 1)) * length - length / 2.0f;
        for (int i = 0; i < segments; ++i) {
            float angle = 2.0f * 3.14159f * i / segments;
            Vertex v1, v2;
            v1.position = glm::vec3(radius * std::cos(angle), radius * std::sin(angle), z);
            v2.position = glm::vec3(radius * std::cos(angle), radius * std::sin(angle), zNext);
            v1.color = glm::vec3(0.5f, 0.5f, 0.5f); // Gray for wire
            v2.color = glm::vec3(0.5f, 0.5f, 0.5f);
            vertices.push_back(v1);
            vertices.push_back(v2);
        }
    }
    return vertices;
}

GLuint createShaderProgram() {
    GLuint vertexShader = glCreateShader(GL_VERTEX_SHADER);
    glShaderSource(vertexShader, 1, &vertexShaderSource, NULL);
    glCompileShader(vertexShader);

    GLint success;
    glGetShaderiv(vertexShader, GL_COMPILE_STATUS, &success);
    if (!success) {
        char infoLog[512];
        glGetShaderInfoLog(vertexShader, 512, NULL, infoLog);
        std::cerr << "Vertex Shader Compilation Failed: " << infoLog << std::endl;
    }

    GLuint fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
    glShaderSource(fragmentShader, 1, &fragmentShaderSource, NULL);
    glCompileShader(fragmentShader);

    glGetShaderiv(fragmentShader, GL_COMPILE_STATUS, &success);
    if (!success) {
        char infoLog[512];
        glGetShaderInfoLog(fragmentShader, 512, NULL, infoLog);
        std::cerr << "Fragment Shader Compilation Failed: " << infoLog << std::endl;
    }

    GLuint program = glCreateProgram();
    glAttachShader(program, vertexShader);
    glAttachShader(program, fragmentShader);
    glLinkProgram(program);

    glGetProgramiv(program, GL_LINK_STATUS, &success);
    if (!success) {
        char infoLog[512];
        glGetProgramInfoLog(program, 512, NULL, infoLog);
        std::cerr << "Shader Program Linking Failed: " << infoLog << std::endl;
    }

    glDeleteShader(vertexShader);
    glDeleteShader(fragmentShader);
    return program;
}

void updateParticles(std::vector<Vertex>& particleVertices, std::vector<float>& particleOffsets, int maxParticles, float current, const std::vector<Vertex>& wireVertices, float length) {
    int numParticles = std::min(maxParticles, std::max(1, (int)(current * 5.0f)));
    particleVertices.resize(numParticles);
    particleOffsets.resize(numParticles);
    const int segments = 20;
    const int lengthSegments = 50;
    for (int i = 0; i < numParticles; ++i) {
        particleOffsets[i] = i * (lengthSegments - 1) / (float)numParticles;
        int idx = (int)(particleOffsets[i]) * 2 * segments; // Adjust for cylinder vertex pairs
        particleVertices[i].position = wireVertices[idx].position;
        particleVertices[i].color = glm::vec3(1.0f, 1.0f, 0.0f); // Yellow for electrons
    }
}

int main() {
    if (!glfwInit()) {
        std::cerr << "Failed to initialize GLFW" << std::endl;
        return -1;
    }

    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    GLFWwindow* window = glfwCreateWindow(800, 600, "Ohm's Law Simulation", NULL, NULL);
    if (!window) {
        std::cerr << "Failed to create GLFW window" << std::endl;
        glfwTerminate();
        return -1;
    }

    glfwMakeContextCurrent(window);
    glewExperimental = GL_TRUE;
    if (glewInit() != GLEW_OK) {
        std::cerr << "Failed to initialize GLEW" << std::endl;
        glfwTerminate();
        return -1;
    }

    glEnable(GL_DEPTH_TEST);
    glViewport(0, 0, 800, 600);

    // Initialize ImGui
    IMGUI_CHECKVERSION();
    ImGui::CreateContext();
    ImGuiIO& io = ImGui::GetIO(); (void)io;
    ImGui::StyleColorsDark();
    ImGui_ImplGlfw_InitForOpenGL(window, true);
    ImGui_ImplOpenGL3_Init("#version 330");

    GLuint shaderProgram = createShaderProgram();
    glUseProgram(shaderProgram);

    // Wire parameters
    float radius = 0.1f; // Initial radius
    float length = 4.0f; // Initial length
    std::vector<Vertex> wireVertices = createWirePath(radius, length);
    GLuint wireVAO, wireVBO;
    glGenVertexArrays(1, &wireVAO);
    glGenBuffers(1, &wireVBO);
    glBindVertexArray(wireVAO);
    glBindBuffer(GL_ARRAY_BUFFER, wireVBO);
    glBufferData(GL_ARRAY_BUFFER, wireVertices.size() * sizeof(Vertex), wireVertices.data(), GL_STATIC_DRAW);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)0);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)(sizeof(glm::vec3)));
    glEnableVertexAttribArray(1);

    // Particle setup
    const int maxParticles = 50;
    float voltage = 5.0f;
    float resistance = 5.0f;
    float current = voltage / resistance;
    std::vector<Vertex> particleVertices;
    std::vector<float> particleOffsets;
    updateParticles(particleVertices, particleOffsets, maxParticles, current, wireVertices, length);

    GLuint particleVAO, particleVBO;
    glGenVertexArrays(1, &particleVAO);
    glGenBuffers(1, &particleVBO);
    glBindVertexArray(particleVAO);
    glBindBuffer(GL_ARRAY_BUFFER, particleVBO);
    glBufferData(GL_ARRAY_BUFFER, maxParticles * sizeof(Vertex), nullptr, GL_DYNAMIC_DRAW);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)0);
    glEnableVertexAttribArray(0);
    glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)(sizeof(glm::vec3)));
    glEnableVertexAttribArray(1);

    GLint mvpLoc = glGetUniformLocation(shaderProgram, "mvp");
    float rotX = 0.0f, rotY = 0.0f, rotZ = 0.0f;
    float particleTime = 0.0f;

    while (!glfwWindowShouldClose(window)) {
        glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

        // Start ImGui frame
        ImGui_ImplOpenGL3_NewFrame();
        ImGui_ImplGlfw_NewFrame();
        ImGui::NewFrame();

        // UI window for controls
        ImGui::Begin("Ohm's Law Controls");
        ImGui::SliderFloat("Voltage (V)", &voltage, 0.0f, 10.0f, "%.1f V");
        ImGui::SliderFloat("Resistance (R)", &resistance, 1.0f, 10.0f, "%.1f Ω");
        current = voltage / std::max(resistance, 1.0f);
        ImGui::Text("Current (I) = %.2f A", current);
        ImGui::SliderFloat("Cylinder Radius", &radius, 0.05f, 0.5f, "%.2f");
        ImGui::SliderFloat("Cylinder Length", &length, 1.0f, 17.0f, "%.1f");
        ImGui::SliderFloat("Rotate X", &rotX, 0.0f, 360.0f, "%.1f°");
        ImGui::SliderFloat("Rotate Y", &rotY, 0.0f, 360.0f, "%.1f°");
        ImGui::SliderFloat("Rotate Z", &rotZ, 0.0f, 360.0f, "%.1f°");
        ImGui::End();

        // Update wire geometry if radius or length changes
        static float prevRadius = radius, prevLength = length;
        if (radius != prevRadius || length != prevLength) {
            wireVertices = createWirePath(radius, length);
            glBindBuffer(GL_ARRAY_BUFFER, wireVBO);
            glBufferData(GL_ARRAY_BUFFER, wireVertices.size() * sizeof(Vertex), wireVertices.data(), GL_STATIC_DRAW);
            prevRadius = radius;
            prevLength = length;
        }

        // Update particles based on current
        updateParticles(particleVertices, particleOffsets, maxParticles, current, wireVertices, length);
        glBindBuffer(GL_ARRAY_BUFFER, particleVBO);
        glBufferData(GL_ARRAY_BUFFER, maxParticles * sizeof(Vertex), nullptr, GL_DYNAMIC_DRAW);
        glBufferSubData(GL_ARRAY_BUFFER, 0, particleVertices.size() * sizeof(Vertex), particleVertices.data());

        // Update MVP matrix with manual rotations
        glm::mat4 model = glm::mat4(1.0f);
        model = glm::rotate(model, glm::radians(rotX), glm::vec3(1.0f, 0.0f, 0.0f));
        model = glm::rotate(model, glm::radians(rotY), glm::vec3(0.0f, 1.0f, 0.0f));
        model = glm::rotate(model, glm::radians(rotZ), glm::vec3(0.0f, 0.0f, 1.0f));
        glm::mat4 view = glm::lookAt(glm::vec3(0.0f, 3.0f, 10.0f), glm::vec3(0.0f, 0.0f, 0.0f), glm::vec3(0.0f, 1.0f, 0.0f));
        glm::mat4 projection = glm::perspective(glm::radians(45.0f), 800.0f / 600.0f, 0.1f, 100.0f);
        glm::mat4 mvp = projection * view * model;
        glUniformMatrix4fv(mvpLoc, 1, GL_FALSE, glm::value_ptr(mvp));

        // Draw wire (cylinder)
        glBindVertexArray(wireVAO);
        const int segments = 20;
        for (int i = 0; i < wireVertices.size() / (2 * segments); ++i) {
            glDrawArrays(GL_TRIANGLE_STRIP, i * 2 * segments, 2 * segments);
        }

        // Draw particles
        float speed = voltage * 0.005f;
        particleTime += speed;
        const int lengthSegments = 50;
        for (size_t i = 0; i < particleVertices.size(); ++i) {
            float t = fmod(particleOffsets[i] + particleTime, lengthSegments - 1);
            int idx = (int)t * 2 * segments;
            float frac = t - (int)t;
            glm::vec3 pos = wireVertices[idx].position;
            if (idx + 2 * segments < wireVertices.size()) {
                pos += frac * (wireVertices[idx + 2 * segments].position - wireVertices[idx].position);
            }
            particleVertices[i].position = pos;
        }

        glBindBuffer(GL_ARRAY_BUFFER, particleVBO);
        glBufferSubData(GL_ARRAY_BUFFER, 0, particleVertices.size() * sizeof(Vertex), particleVertices.data());
        glBindVertexArray(particleVAO);
        glPointSize(10.0f);
        glDrawArrays(GL_POINTS, 0, particleVertices.size());

        // Render ImGui
        ImGui::Render();
        ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());

        glfwSwapBuffers(window);
        glfwPollEvents();
    }

    // Cleanup ImGui
    ImGui_ImplOpenGL3_Shutdown();
    ImGui_ImplGlfw_Shutdown();
    ImGui::DestroyContext();

    // Cleanup OpenGL
    glDeleteVertexArrays(1, &wireVAO);
    glDeleteVertexArrays(1, &particleVAO);
    glDeleteBuffers(1, &wireVBO);
    glDeleteBuffers(1, &particleVBO);
    glDeleteProgram(shaderProgram);

    glfwTerminate();
    return 0;
}
